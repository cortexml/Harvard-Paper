tmp = urr.copy()
tmp = tmp[tmp['Plan ID (Standard Component ID)'].map(lambda p:p in ffmPlanIds)]

res = ''
marketLookup = {}
compLookup = {}
rankLookup = {}
for s,g in tmp.groupby('State'):
    if len(set(g['HIOS Issuer ID']))>=1:
        marketLookup[s]={}
        compLookup[s]={}
        rankLookup[s]={}
        line = s+','
        g = g.drop_duplicates(subset=['HIOS Issuer ID']).\
            sort('Projected Member Months',ascending=False)
        g['MarketSize'] = g['Projected Member Months']/sum(g['Projected Member Months'])*100
        rank = 1
        for _,r in g.sort('MarketSize',ascending=False).iterrows():
            line += '"'+r['Company Legal Name']+'",'+str(r['MarketSize'])+','
            marketLookup[s][r['HIOS Issuer ID']]=r['MarketSize']
            compLookup[s]=len(set(g['HIOS Issuer ID']))
            rankLookup[s][r['HIOS Issuer ID']] = rank
            rank+=1
        res+=line+'\n'
